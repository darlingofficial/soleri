```ts
// packages/core/src/cognee/sync-manager.ts
import { createHash } from 'node:crypto';
import type { PersistenceProvider } from '../persistence/index.js';
import type { CogneeClient } from './client.js';
import type { IntelligenceEntry } from '../intelligence/types.js';

export type SyncOp = 'ingest' | 'update' | 'delete';
export type SyncStatus = 'pending' | 'processing' | 'completed' | 'failed';

export interface SyncQueueItem {
  id: number;
  op: SyncOp;
  entryId: string;
  dataset: string;
  contentHash: string | null;
  status: SyncStatus;
  attempts: number;
  error: string | null;
  createdAt: number;
  processedAt: number | null;
}

export interface SyncManagerStats {
  pending: number;
  processing: number;
  completed: number;
  failed: number;
  lastDrainAt: number | null;
  queueSize: number;
}

const BATCH_SIZE = 10;
const MAX_RETRIES = 3;

export class CogneeSyncManager {
  private provider: PersistenceProvider;
  private cognee: CogneeClient;
  private dataset: string;
  private lastDrainAt: number | null = null;
  private drainTimer: ReturnType<typeof setInterval> | null = null;
  private previouslyAvailable = false;

  constructor(provider: PersistenceProvider, cognee: CogneeClient, dataset: string) {
    this.provider = provider;
    this.cognee = cognee;
    this.dataset = dataset;
    this.initSchema();
  }

  private initSchema(): void {
    this.provider.execSql(`
      CREATE TABLE IF NOT EXISTS cognee_sync_queue (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        op TEXT NOT NULL,
        entry_id TEXT NOT NULL,
        dataset TEXT NOT NULL,
        content_hash TEXT,
        status TEXT NOT NULL DEFAULT 'pending',
        attempts INTEGER NOT NULL DEFAULT 0,
        error TEXT,
        created_at INTEGER NOT NULL DEFAULT (unixepoch()),
        processed_at INTEGER
      );
    `);
    // Add cognee_ingested_hash column to entries if missing
    try {
      this.provider.run('ALTER TABLE entries ADD COLUMN cognee_ingested_hash TEXT');
    } catch {
      // Column already exists
    }
  }

  /** Compute a content hash for an entry (SHA-256 of serialized fields). */
  static contentHash(entry: IntelligenceEntry): string {
    const text = [entry.title, entry.description, entry.context, entry.tags?.join(',')].join('|');
    return createHash('sha256').update(text).digest('hex').slice(0, 16);
  }

  /** Enqueue a sync operation. Called by Vault on seed/update/remove. */
  enqueue(op: SyncOp, entryId: string, entry?: IntelligenceEntry): void {
    const hash = entry ? CogneeSyncManager.contentHash(entry) : null;
    this.provider.run(
      `INSERT INTO cognee_sync_queue (op, entry_id, dataset, content_hash, created_at)
       VALUES (@op, @entryId, @dataset, @hash, unixepoch())`,
      { op, entryId, dataset: this.dataset, hash },
    );
  }

  /** Drain pending items to Cognee. Returns count of successfully processed items. */
  async drain(): Promise<number> {
    if (!this.cognee.isAvailable) return 0;

    const pending = this.provider.all<Record<string, unknown>>(
      `SELECT * FROM cognee_sync_queue WHERE status = 'pending' ORDER BY created_at ASC LIMIT @limit`,
      { limit: BATCH_SIZE },
    );

    if (pending.length === 0) return 0;

    let processed = 0;
    for (const row of pending) {
      const id = row.id as number;
      const op = row.op as SyncOp;
      const entryId = row.entry_id as string;
      const hash = row.content_hash as string | null;
      const attempts = (row.attempts as number) + 1;

      this.provider.run(
        `UPDATE cognee_sync_queue SET status = 'processing', attempts = @attempts WHERE id = @id`,
        { id, attempts },
      );

      try {
        if (op === 'delete') {
          await this.cognee.deleteEntries([entryId]);
        } else {
          // ingest or update — re-read entry from vault and push to Cognee
          const entry = this.provider.get<Record<string, unknown>>(
            'SELECT * FROM entries WHERE id = @id',
            { id: entryId },
          );
          if (entry) {
            await this.cognee.addEntries([this.rowToEntry(entry)]);
            // Update cognee_ingested_hash on success
            if (hash) {
              this.provider.run(
                'UPDATE entries SET cognee_ingested_hash = @hash WHERE id = @id',
                { hash, id: entryId },
              );
            }
          }
        }
        this.provider.run(
          `UPDATE cognee_sync_queue SET status = 'completed', processed_at = unixepoch() WHERE id = @id`,
          { id },
        );
        processed++;
      } catch (err) {
        const error = err instanceof Error ? err.message : String(err);
        const newStatus = attempts >= MAX_RETRIES ? 'failed' : 'pending';
        this.provider.run(
          `UPDATE cognee_sync_queue SET status = @status, error = @error WHERE id = @id`,
          { id, status: newStatus, error },
        );
      }
    }

    this.lastDrainAt = Date.now();
    return processed;
  }

  /** Reconciliation: find entries where content_hash != cognee_ingested_hash and enqueue. */
  reconcile(): number {
    const dirty = this.provider.all<{ id: string }>(
      `SELECT id FROM entries WHERE cognee_ingested_hash IS NULL
       OR cognee_ingested_hash != substr(
         hex(zeroblob(0)), 1, 0
       )`,
    );
    // Simpler: just find entries not yet synced or with stale hash
    const entries = this.provider.all<Record<string, unknown>>(
      `SELECT * FROM entries WHERE cognee_ingested_hash IS NULL`,
    );
    let enqueued = 0;
    for (const row of entries) {
      const entry = this.rowToEntry(row);
      const hash = CogneeSyncManager.contentHash(entry);
      this.enqueue('ingest', entry.id, entry);
      enqueued++;
    }
    // Also find entries where hash doesn't match
    const stale = this.provider.all<Record<string, unknown>>(
      `SELECT * FROM entries WHERE cognee_ingested_hash IS NOT NULL`,
    );
    for (const row of stale) {
      const entry = this.rowToEntry(row);
      const hash = CogneeSyncManager.contentHash(entry);
      const existing = row.cognee_ingested_hash as string;
      if (existing !== hash) {
        this.enqueue('update', entry.id, entry);
        enqueued++;
      }
    }
    return enqueued;
  }

  /** Get queue statistics. */
  getStats(): SyncManagerStats {
    const counts = this.provider.get<Record<string, number>>(
      `SELECT
        SUM(CASE WHEN status = 'pending' THEN 1 ELSE 0 END) as pending,
        SUM(CASE WHEN status = 'processing' THEN 1 ELSE 0 END) as processing,
        SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) as completed,
        SUM(CASE WHEN status = 'failed' THEN 1 ELSE 0 END) as failed,
        COUNT(*) as total
       FROM cognee_sync_queue`,
    );
    return {
      pending: counts?.pending ?? 0,
      processing: counts?.processing ?? 0,
      completed: counts?.completed ?? 0,
      failed: counts?.failed ?? 0,
      lastDrainAt: this.lastDrainAt,
      queueSize: counts?.total ?? 0,
    };
  }

  /** Check for health-flip and auto-drain. */
  async checkHealthFlip(): Promise<boolean> {
    const available = this.cognee.isAvailable;
    if (available && !this.previouslyAvailable) {
      this.previouslyAvailable = true;
      await this.drain();
      return true;
    }
    this.previouslyAvailable = available;
    return false;
  }

  /** Convert a raw DB row to IntelligenceEntry. */
  private rowToEntry(row: Record<string, unknown>): IntelligenceEntry {
    return {
      id: row.id as string,
      type: row.type as IntelligenceEntry['type'],
      domain: row.domain as string,
      title: row.title as string,
      severity: row.severity as IntelligenceEntry['severity'],
      description: row.description as string,
      context: (row.context as string) ?? undefined,
      example: (row.example as string) ?? undefined,
      counterExample: (row.counter_example as string) ?? undefined,
      why: (row.why as string) ?? undefined,
      tags: JSON.parse((row.tags as string) || '[]'),
      appliesTo: JSON.parse((row.applies_to as string) || '[]'),
    };
  }

  /** Stop any drain timer. */
  close(): void {
    if (this.drainTimer) {
      clearInterval(this.drainTimer);
      this.drainTimer = null;
    }
  }
}
```

```ts
import type { OpDefinition } from '../facades/types.js';
import type { CogneeSyncManager } from '../cognee/sync-manager.js';

export function createCogneeSyncOps(syncManager: CogneeSyncManager | null): OpDefinition[] {
  return [
    {
      name: 'cognee_sync_status',
      description: 'Queue stats: pending/processing/failed counts, last drain timestamp.',
      auth: 'read',
      schema: z.object({}),
      handler: async () => {
        if (!syncManager) return { error: 'Sync manager not configured' };
        return syncManager.getStats();
      },
    },
    {
      name: 'cognee_sync_drain',
      description: 'Force immediate drain of pending sync queue to Cognee.',
      auth: 'write',
      schema: z.object({}),
      handler: async () => {
        if (!syncManager) return { error: 'Sync manager not configured' };
        const processed = await syncManager.drain();
        return { processed, stats: syncManager.getStats() };
      },
    },
    {
      name: 'cognee_sync_reconcile',
      description: 'Full reconciliation: scan entries where content differs from last Cognee sync, enqueue dirty ones.',
      auth: 'write',
      schema: z.object({}),
      handler: async () => {
        if (!syncManager) return { error: 'Sync manager not configured' };
        const enqueued = syncManager.reconcile();
        return { enqueued, stats: syncManager.getStats() };
      },
    },
  ];
}
```

```ts
// packages/core/src/intake/content-classifier.ts
import type { LLMClient } from '../llm/llm-client.js';
import type { ClassifiedItem, KnowledgeType } from './types.js';

const VALID_TYPES: KnowledgeType[] = [
  'pattern', 'anti-pattern', 'principle', 'concept',
  'reference', 'workflow', 'idea', 'roadmap',
];

const CLASSIFICATION_PROMPT = `You are a knowledge classifier. Given a chunk of text from a technical book, extract discrete knowledge items.

For each item, provide:
- type: one of: pattern, anti-pattern, principle, concept, reference, workflow, idea, roadmap
- title: concise title (max 80 chars)
- description: 2-3 sentence description
- tags: 3-5 relevant lowercase tags
- severity: critical (must-know), warning (should-know), suggestion (nice-to-know)
- citation: chapter/page reference from the source text

Return a JSON array of items. If the chunk has no extractable knowledge, return [].

Respond ONLY with valid JSON — no markdown fences, no explanation.`;

export async function classifyChunk(
  llm: LLMClient,
  chunkText: string,
  citation: string,
): Promise<ClassifiedItem[]> {
  try {
    const result = await llm.call({
      messages: [
        { role: 'system', content: CLASSIFICATION_PROMPT },
        { role: 'user', content: `Source: ${citation}\n\n${chunkText}` },
      ],
      maxTokens: 4096,
    });
    const text = result.content.trim();
    const parsed = JSON.parse(text);
    if (!Array.isArray(parsed)) return [];
    // Validate and sanitize
    return parsed
      .filter(
        (item: Record<string, unknown>) =>
          typeof item.title === 'string' &&
          typeof item.description === 'string' &&
          VALID_TYPES.includes(item.type as KnowledgeType),
      )
      .map((item: Record<string, unknown>) => ({
        type: item.type as KnowledgeType,
        title: String(item.title).slice(0, 80),
        description: String(item.description),
        tags: Array.isArray(item.tags) ? item.tags.map(String).slice(0, 5) : [],
        severity: ['critical', 'warning', 'suggestion'].includes(String(item.severity))
          ? (String(item.severity) as ClassifiedItem['severity'])
          : 'suggestion',
        citation: String(item.citation || citation),
      }));
  } catch {
    return [];
  }
}
```

```ts
// packages/core/src/intake/dedup-gate.ts
import { tokenize, calculateTfIdf, cosineSimilarity, type SparseVector } from '../text/similarity.js';
import type { Vault } from '../vault/vault.js';
import type { ClassifiedItem } from './types.js';

const DEDUP_THRESHOLD = 0.85;

export interface DedupResult {
  item: ClassifiedItem;
  isDuplicate: boolean;
  bestMatchId?: string;
  similarity: number;
}

/**
 * Check classified items against existing vault entries for duplicates.
 * Uses TF-IDF cosine similarity (same infra as Curator).
 */
export function dedupItems(items: ClassifiedItem[], vault: Vault): DedupResult[] {
  // Build vocabulary from existing vault entries
  const allEntries = vault.listAll();
  if (allEntries.length === 0) {
    return items.map((item) => ({ item, isDuplicate: false, similarity: 0 }));
  }

  // Build IDF from all existing entries + new items
  const docFreq = new Map<string, number>();
  const totalDocs = allEntries.length + items.length;

  const entryTexts = allEntries.map((e) => `${e.title} ${e.description} ${e.context ?? ''}`);
  const itemTexts = items.map((i) => `${i.title} ${i.description}`);
  const allTexts = [...entryTexts, ...itemTexts];

  for (const text of allTexts) {
    const terms = new Set(tokenize(text));
    for (const term of terms) {
      docFreq.set(term, (docFreq.get(term) ?? 0) + 1);
    }
  }

  // Build IDF map
  const vocabulary = new Map<string, number>();
  for (const [term, df] of docFreq) {
    vocabulary.set(term, Math.log((totalDocs + 1) / (df + 1)) + 1);
  }

  // Build vectors for existing entries
  const entryVectors: Array<{ id: string; vec: SparseVector }> = allEntries.map((e, i) => ({
    id: e.id,
    vec: calculateTfIdf(tokenize(entryTexts[i]), vocabulary),
  }));

  return items.map((item) => {
    const text = `${item.title} ${item.description}`;
    const vec = calculateTfIdf(tokenize(text), vocabulary);

    let bestSim = 0;
    let bestId: string | undefined;

    for (const { id, vec: entryVec } of entryVectors) {
      const sim = cosineSimilarity(vec, entryVec);
      if (sim > bestSim) {
        bestSim = sim;
        bestId = id;
      }
    }

    return {
      item,
      isDuplicate: bestSim >= DEDUP_THRESHOLD,
      bestMatchId: bestSim >= DEDUP_THRESHOLD ? bestId : undefined,
      similarity: bestSim,
    };
  });
}
```

```ts
// packages/core/src/intake/intake-pipeline.ts
import { createHash, randomUUID } from 'node:crypto';
import { readFileSync, statSync } from 'node:fs';
import type { PersistenceProvider } from '../persistence/index.js';
import type { Vault } from '../vault/vault.js';
import type { LLMClient } from '../llm/llm-client.js';
import type {
  IntakeConfig,
  IntakeJobRecord,
  IntakeChunk,
  IntakeJobStatus,
  ClassifiedItem,
  IntakePreviewResult,
} from './types.js';
import { classifyChunk } from './content-classifier.js';
import { dedupItems } from './dedup-gate.js';

const DEFAULT_CHUNK_SIZE = 10; // pages per chunk

export class IntakePipeline {
  private provider: PersistenceProvider;
  private vault: Vault;
  private llm: LLMClient;

  constructor(provider: PersistenceProvider, vault: Vault, llm: LLMClient) {
    this.provider = provider;
    this.vault = vault;
    this.llm = llm;
    this.initSchema();
  }

  private initSchema(): void {
    this.provider.execSql(`
      CREATE TABLE IF NOT EXISTS intake_jobs (
        id TEXT PRIMARY KEY,
        status TEXT NOT NULL,
        config TEXT NOT NULL,
        pdf_meta TEXT,
        toc TEXT,
        stats TEXT,
        created_at INTEGER NOT NULL DEFAULT (unixepoch()),
        updated_at INTEGER NOT NULL DEFAULT (unixepoch()),
        completed_at INTEGER
      );
      CREATE TABLE IF NOT EXISTS intake_chunks (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        job_id TEXT NOT NULL REFERENCES intake_jobs(id),
        chunk_index INTEGER NOT NULL,
        title TEXT,
        page_start INTEGER NOT NULL,
        page_end INTEGER NOT NULL,
        status TEXT NOT NULL DEFAULT 'pending',
        items_extracted INTEGER DEFAULT 0,
        items_stored INTEGER DEFAULT 0,
        items_deduped INTEGER DEFAULT 0,
        error TEXT,
        processed_at INTEGER
      );
    `);
  }

  /** Stage 1: Initialize — parse PDF, create job, create chunks. */
  async ingestBook(config: IntakeConfig): Promise<IntakeJobRecord> {
    const jobId = randomUUID().slice(0, 8);
    const chunkSize = config.chunkPageSize ?? DEFAULT_CHUNK_SIZE;

    // Parse PDF using pdf-parse (dynamic import — optional dependency)
    let pdfData: { text: string; numpages: number; info: Record<string, unknown> };
    try {
      const pdfParse = (await import('pdf-parse')).default;
      const buffer = readFileSync(config.pdfPath);
      pdfData = await pdfParse(buffer);
    } catch (err) {
      throw new Error(`PDF parse failed: ${err instanceof Error ? err.message : String(err)}`);
    }

    const fileStats = statSync(config.pdfPath);
    const fileHash = createHash('sha256')
      .update(readFileSync(config.pdfPath))
      .digest('hex')
      .slice(0, 16);

    const pdfMeta = {
      totalPages: pdfData.numpages,
      fileHash,
      fileSize: fileStats.size,
    };

    // Create chunks (fixed N-page chunks)
    const totalPages = pdfData.numpages;
    const chunks: Array<{ index: number; pageStart: number; pageEnd: number }> = [];
    for (let start = 1; start <= totalPages; start += chunkSize) {
      const end = Math.min(start + chunkSize - 1, totalPages);
      chunks.push({ index: chunks.length, pageStart: start, pageEnd: end });
    }

    // Persist job
    this.provider.run(
      `INSERT INTO intake_jobs (id, status, config, pdf_meta, created_at, updated_at)
       VALUES (@id, 'initialized', @config, @pdfMeta, unixepoch(), unixepoch())`,
      {
        id: jobId,
        config: JSON.stringify(config),
        pdfMeta: JSON.stringify(pdfMeta),
      },
    );

    // Persist chunks
    for (const chunk of chunks) {
      this.provider.run(
        `INSERT INTO intake_chunks (job_id, chunk_index, page_start, page_end, status)
         VALUES (@jobId, @index, @start, @end, 'pending')`,
        { jobId, index: chunk.index, start: chunk.pageStart, end: chunk.pageEnd },
      );
    }

    // Store page text for later processing (in memory via closure,
    // but for multi-call we'll re-parse — tradeoff for simplicity)
    return this.getJob(jobId)!;
  }

  /** Stage 2-5: Process next N pending chunks of a job. */
  async processChunks(jobId: string, count: number = 5): Promise<{
    processed: number;
    itemsStored: number;
    itemsDeduped: number;
    remaining: number;
  }> {
    const job = this.getJob(jobId);
    if (!job) throw new Error(`Job ${jobId} not found`);
    if (job.status === 'completed' || job.status === 'cancelled') {
      return { processed: 0, itemsStored: 0, itemsDeduped: 0, remaining: 0 };
    }

    // Update job status
    this.provider.run(
      `UPDATE intake_jobs SET status = 'processing', updated_at = unixepoch() WHERE id = @id`,
      { id: jobId },
    );

    // Re-parse PDF to get text
    const config = job.config;
    let pdfData: { text: string; numpages: number };
    try {
      const pdfParse = (await import('pdf-parse')).default;
      const buffer = readFileSync(config.pdfPath);
      pdfData = await pdfParse(buffer);
    } catch (err) {
      this.provider.run(
        `UPDATE intake_jobs SET status = 'failed', updated_at = unixepoch() WHERE id = @id`,
        { id: jobId },
      );
      throw new Error(`PDF re-parse failed: ${err instanceof Error ? err.message : String(err)}`);
    }

    // Split text into pages (approximate: split by form feeds or equal chunks)
    const pages = splitIntoPages(pdfData.text, pdfData.numpages);

    // Get pending chunks
    const pendingChunks = this.provider.all<Record<string, unknown>>(
      `SELECT * FROM intake_chunks WHERE job_id = @jobId AND status = 'pending'
       ORDER BY chunk_index ASC LIMIT @count`,
      { jobId, count },
    );

    let totalStored = 0;
    let totalDeduped = 0;
    let processed = 0;

    for (const row of pendingChunks) {
      const chunkId = row.id as number;
      const pageStart = row.page_start as number;
      const pageEnd = row.page_end as number;
      const chunkIndex = row.chunk_index as number;

      this.provider.run(
        `UPDATE intake_chunks SET status = 'processing' WHERE id = @id`,
        { id: chunkId },
      );

      try {
        // Extract text for this chunk's page range
        const chunkText = pages.slice(pageStart - 1, pageEnd).join('\n\n');
        const citation = `${config.title}, pp. ${pageStart}-${pageEnd}`;

        // Stage 3: Classify via LLM
        const items = await classifyChunk(this.llm, chunkText, citation);

        // Stage 4: Dedup against vault
        const dedupResults = dedupItems(items, this.vault);
        const unique = dedupResults.filter((r) => !r.isDuplicate);
        const deduped = dedupResults.filter((r) => r.isDuplicate).length;

        // Stage 5: Store unique items in vault
        let stored = 0;
        for (const result of unique) {
          const entryId = `intake-${jobId}-${chunkIndex}-${stored}`;
          this.vault.add({
            id: entryId,
            type: result.item.type === 'anti-pattern' ? 'anti-pattern' : result.item.type === 'pattern' ? 'pattern' : 'rule',
            domain: config.domain,
            title: result.item.title,
            severity: result.item.severity,
            description: result.item.description,
            context: result.item.citation,
            tags: [...(config.tags ?? []), ...result.item.tags],
          });
          stored++;
        }

        this.provider.run(
          `UPDATE intake_chunks SET status = 'completed', items_extracted = @extracted,
           items_stored = @stored, items_deduped = @deduped, processed_at = unixepoch()
           WHERE id = @id`,
          { id: chunkId, extracted: items.length, stored, deduped },
        );

        totalStored += stored;
        totalDeduped += deduped;
        processed++;
      } catch (err) {
        this.provider.run(
          `UPDATE intake_chunks SET status = 'failed', error = @error WHERE id = @id`,
          { id: chunkId, error: err instanceof Error ? err.message : String(err) },
        );
      }
    }

    // Check if all chunks are done
    const remaining = this.provider.get<{ cnt: number }>(
      `SELECT COUNT(*) as cnt FROM intake_chunks WHERE job_id = @jobId AND status = 'pending'`,
      { jobId },
    )?.cnt ?? 0;

    // Stage 6: Finalize if all done
    if (remaining === 0) {
      this.finalizeJob(jobId);
    }

    return { processed, itemsStored: totalStored, itemsDeduped: totalDeduped, remaining };
  }

  /** Preview classification on a page range without storing. */
  async preview(config: IntakeConfig, pageStart: number, pageEnd: number): Promise<IntakePreviewResult> {
    const pdfParse = (await import('pdf-parse')).default;
    const buffer = readFileSync(config.pdfPath);
    const pdfData = await pdfParse(buffer);
    const pages = splitIntoPages(pdfData.text, pdfData.numpages);
    const chunkText = pages.slice(pageStart - 1, pageEnd).join('\n\n');
    const citation = `${config.title}, pp. ${pageStart}-${pageEnd}`;
    const items = await classifyChunk(this.llm, chunkText, citation);
    return { items, chunkText: chunkText.slice(0, 2000), pageRange: { start: pageStart, end: pageEnd } };
  }

  /** Get a job by ID. */
  getJob(jobId: string): IntakeJobRecord | null {
    const row = this.provider.get<Record<string, unknown>>(
      `SELECT * FROM intake_jobs WHERE id = @id`,
      { id: jobId },
    );
    if (!row) return null;
    return {
      id: row.id as string,
      status: row.status as IntakeJobStatus,
      config: JSON.parse(row.config as string),
      pdfMeta: row.pdf_meta ? JSON.parse(row.pdf_meta as string) : null,
      toc: row.toc ? JSON.parse(row.toc as string) : null,
      stats: row.stats ? JSON.parse(row.stats as string) : null,
      createdAt: row.created_at as number,
      updatedAt: row.updated_at as number,
      completedAt: row.completed_at ? (row.completed_at as number) : null,
    };
  }

  /** List all jobs. */
  listJobs(): IntakeJobRecord[] {
    const rows = this.provider.all<Record<string, unknown>>(
      `SELECT * FROM intake_jobs ORDER BY created_at DESC`,
    );
    return rows.map((row) => ({
      id: row.id as string,
      status: row.status as IntakeJobStatus,
      config: JSON.parse(row.config as string),
      pdfMeta: row.pdf_meta ? JSON.parse(row.pdf_meta as string) : null,
      toc: row.toc ? JSON.parse(row.toc as string) : null,
      stats: row.stats ? JSON.parse(row.stats as string) : null,
      createdAt: row.created_at as number,
      updatedAt: row.updated_at as number,
      completedAt: row.completed_at ? (row.completed_at as number) : null,
    }));
  }

  /** Get chunks for a job. */
  getChunks(jobId: string): IntakeChunk[] {
    return this.provider.all<Record<string, unknown>>(
      `SELECT * FROM intake_chunks WHERE job_id = @jobId ORDER BY chunk_index`,
      { jobId },
    ).map((row) => ({
      id: row.id as number,
      jobId: row.job_id as string,
      chunkIndex: row.chunk_index as number,
      title: (row.title as string) ?? null,
      pageStart: row.page_start as number,
      pageEnd: row.page_end as number,
      status: row.status as IntakeChunk['status'],
      itemsExtracted: (row.items_extracted as number) ?? 0,
      itemsStored: (row.items_stored as number) ?? 0,
      itemsDeduped: (row.items_deduped as number) ?? 0,
      error: (row.error as string) ?? null,
      processedAt: (row.processed_at as number) ?? null,
    }));
  }

  /** Stage 6: Finalize — compute stats, mark complete. */
  private finalizeJob(jobId: string): void {
    const chunks = this.getChunks(jobId);
    const stats = {
      itemsExtracted: chunks.reduce((s, c) => s + c.itemsExtracted, 0),
      itemsStored: chunks.reduce((s, c) => s + c.itemsStored, 0),
      itemsDeduped: chunks.reduce((s, c) => s + c.itemsDeduped, 0),
      itemsFailed: chunks.filter((c) => c.status === 'failed').length,
    };
    const hasFailed = chunks.some((c) => c.status === 'failed');
    this.provider.run(
      `UPDATE intake_jobs SET status = @status, stats = @stats,
       completed_at = unixepoch(), updated_at = unixepoch() WHERE id = @id`,
      {
        id: jobId,
        status: hasFailed ? 'completed' : 'completed', // completed even with partial failures
        stats: JSON.stringify(stats),
      },
    );
  }
}

/** Split PDF text into approximate page-sized chunks. */
function splitIntoPages(text: string, numPages: number): string[] {
  // Try form feed splits first
  const ffPages = text.split('\f');
  if (ffPages.length >= numPages * 0.8) {
    return ffPages;
  }
  // Fall back to equal-length splits
  const chunkSize = Math.ceil(text.length / numPages);
  const pages: string[] = [];
  for (let i = 0; i < text.length; i += chunkSize) {
    pages.push(text.slice(i, i + chunkSize));
  }
  return pages;
}
```

```ts
/**
 * Intake operations — 4 ops for book/PDF ingestion.

import { z } from 'zod';
import type { OpDefinition } from '../facades/types.js';
import type { IntakePipeline } from '../intake/intake-pipeline.js';

export function createIntakeOps(pipeline: IntakePipeline | null): OpDefinition[] {
  return [
    {
      name: 'intake_ingest_book',
      description: 'Initialize an intake job from a PDF file path. Creates chunks and prepares for processing.',
      auth: 'write',
      schema: z.object({
        pdfPath: z.string().describe('Absolute path to PDF file'),
        title: z.string().describe('Book title'),
        domain: z.string().describe('Domain to file entries under'),
        author: z.string().optional().describe('Book author'),
        chunkPageSize: z.number().optional().describe('Pages per chunk (default: 10)'),
        tags: z.array(z.string()).optional().describe('Tags to apply to all extracted entries'),
      }),
      handler: async (params) => {
        if (!pipeline) return { error: 'Intake pipeline not configured' };
        return pipeline.ingestBook(params as Parameters<IntakePipeline['ingestBook']>[0]);
      },
    },
    {
      name: 'intake_process',
      description: 'Process next N chunks of an intake job (classify, dedup, store).',
      auth: 'write',
      schema: z.object({
        jobId: z.string().describe('Intake job ID'),
        count: z.number().optional().describe('Number of chunks to process (default: 5)'),
      }),
      handler: async (params) => {
        if (!pipeline) return { error: 'Intake pipeline not configured' };
        const { jobId, count } = params as { jobId: string; count?: number };
        return pipeline.processChunks(jobId, count);
      },
    },
    {
      name: 'intake_status',
      description: 'Get job progress or list all intake jobs.',
      auth: 'read',
      schema: z.object({
        jobId: z.string().optional().describe('Specific job ID, or omit to list all'),
      }),
      handler: async (params) => {
        if (!pipeline) return { error: 'Intake pipeline not configured' };
        const { jobId } = params as { jobId?: string };
        if (jobId) {
          const job = pipeline.getJob(jobId);
          if (!job) return { error: `Job ${jobId} not found` };
          const chunks = pipeline.getChunks(jobId);
          return { job, chunks };
        }
        return { jobs: pipeline.listJobs() };
      },
    },
    {
      name: 'intake_preview',
      description: 'Preview classification on a page range without storing anything.',
      auth: 'read',
      schema: z.object({
        pdfPath: z.string().describe('Absolute path to PDF file'),
        title: z.string().describe('Book title'),
        domain: z.string().describe('Domain for classification context'),
        pageStart: z.number().describe('First page (1-indexed)'),
        pageEnd: z.number().describe('Last page (inclusive)'),
      }),
      handler: async (params) => {
        if (!pipeline) return { error: 'Intake pipeline not configured' };
        const { pdfPath, title, domain, pageStart, pageEnd } = params as {
          pdfPath: string; title: string; domain: string; pageStart: number; pageEnd: number;
        };
        return pipeline.preview({ pdfPath, title, domain }, pageStart, pageEnd);
      },
    },
  ];
}
```
